\subsection{Utility classes}

% Validation
\static{Validation}{validation}
This class provides static methods for checking the validity of given values. \\

\vspace{.5cm}
\hrule

\paragraph*{Method Summary}
\paragraph*{}
\begin{longtable}{Lp{10cm}}
	\startmethodtable

	\method{public static boolean}{isValidFile(File file)}{validation:isvalidfile} \\
	& Checks if the given file is valid. \\
	\method{public static boolean}{isValidPlayerName(String name)}{validation:isvalidplayername} \\
	& Checks if the given string is a valid player name. \\
		
	\hline
\end{longtable}
